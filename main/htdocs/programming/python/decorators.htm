def print_timing(func):
    show_timing = False
    #show_timing = True
    def wrapper(*arg):
        if show_timing: princ("Executing {0}".format(func.func_name))
        t1 = time.time()
        res = func(*arg)
        t2 = time.time()
        if show_timing: princ('%s took %0.3f s' % (func.func_name, (t2-t1)))
        return res
    return wrapper

@print_timing
def foo():
