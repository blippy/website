Python decimals

Working with arbitrary precision arithmetic 
import decimal
decimal.Decimal(5).quantize(decimal.Decimal('0.001')) # Decimal('5.000')
decimal.Decimal(5.000003).quantize(decimal.Decimal('0.001')) # Decimal('5.000')
decimal.Decimal(5.000003).quantize(decimal.Decimal('0.01')) # Decimal('5.00')
decimal.Decimal(5.001).quantize(decimal.Decimal('0.01')) # Decimal('5.00')
decimal.Decimal(5.001).quantize(decimal.Decimal('0.0001')) # Decimal('5.0010')
decimal.Decimal('5.001').quantize(decimal.Decimal('0.0001')) # Decimal('5.0010')
decimal.Decimal(5.0999).quantize(decimal.Decimal('0.1')) # Decimal('5.1')
decimal.Decimal(5.09).quantize(decimal.Decimal('0.1')) # Decimal('5.1')
