Exceptions

Simple try and catch

module Scratch where

import Control.Exception (catch)

main = toTry `catch` handler

toTry :: IO ()
toTry = do
  s <- readFile "README.txt"
  putStrLn s

handler :: IOError -> IO ()
handler e = putStrLn "no can do"
