Building CLISP 2.47 on Ubuntu Ibex

Here I show you how to build some extra goodies for CLISP 
apt-get install libsigsegv-dev   libreadline5-dev  libffcall1-dev
apt-get install python-gtk2-dev libglade2-dev # add some gtk2 goo
apt-get install libgdbm-dev
cd ~/src/clisp/clisp

rm -rf build-mc

./configure  \
	--with-module=gtk2 \
	--with-module=clx/new-clx \
	--with-module=syscalls \
	--with-readline \
	--with-module=gdbm \
	--cbc build-mc

cd build-mc
make
sudo make install
