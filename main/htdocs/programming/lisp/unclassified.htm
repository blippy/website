Macro that simplifies variable definition

(defmacro var (&body body)
 (if (string= '= (second body))
    `(let ((,(first body) ,(third body)))
      (var ,@(cdddr body)))
      `(progn ,@body)))

(var a = 2
    b = (+ a 2)
    c = (* a b)
 (print (list a b c)))
