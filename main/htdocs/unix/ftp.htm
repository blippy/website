Setting up an FTP server with vsftpd 

apt-get install vsftpd
sudo mkdir /var/ftp

-------------------------------------
sudo vi /etc/vsftpd.conf :

# changes: 
#Set anonymous only (xlink)
anonymous_enable=YES
local_enable=NO
write_enable=NO


# add the following:
anon_other_write_enable=NO
anon_world_readable_only=YES
anon_root=/var/ftp
#enable passive mode:
pasv_enable=Yes
pasv_max_port=10100
pasv_min_port=10090

-------------------------------------

sudo service vsftpd restart
