NFS

Setting up a network file system

Server (remote) (cubie)


sudo apt-get install nfs-kernel-server nfs-common

sudo vi /etc/exports:
/home vivo(rw,sync,no_subtree_check)
/ vivo(rw,sync,no_subtree_check)

sudo service nfs-kernel-server restart

Local (vivo)

sudo apt-get install nfs-common
  
# create mount directories:
sudo mkdir -p /mnt/cubie
sudo mkdir -p /mnt/vivo

# temporary mount
sudo mount -t nfs cubie:/ /mnt/cubie
sudo mount -t nfs vivo:/ /mnt/vivo

# permanent mount in /etc/fstab:
cubie:/   /mnt/cubie nfs   rsize=8192,wsize=8192,timeo=14,intr,_netdev   0 0
vivo:/   /mnt/vivo   nfs   rsize=8192,wsize=8192,timeo=14,intr,_netdev   0 0
