NFS

Setting up a network file system

Server (remote) (ernst)


sudo apt-get install nfs-kernel-server nfs-common

sudo vi /etc/exports:
/home miro(rw,sync,no_subtree_check)
/ miro(rw,sync,no_subtree_check)

sudo service nfs-kernel-server restart

Local (miro)

# create mount directories:
sudo mkdir -p /mnt/ernst-home 
sudo mkdir -p /mnt/ernst-root

# temporary mount
sudo mount -t nfs ernst:/home /mnt/ernst-home
sudo mount -t nfs ernst:/ /mnt/ernst-root

# permanent mount in /etc/fstab:
ernst:/home   /mnt/ernst-home   nfs   rsize=8192,wsize=8192,timeo=14,intr,_netdev       0 0
ernst:/   /mnt/ernst-root   nfs   rsize=8192,wsize=8192,timeo=14,intr,_netdev       0 0
