keyboard

nano /etc/locale.gen # uncomment all en_GB* for Britain
locale-gen

Network

  ip addr show # show your IP address
  ip link set enp3s0 down # take down ethernet interface

timezone for UK

  cd /usr/share/zoneinfo/GB
  ls
  ln -s /usr/share/zoneinfo/Europe/London /etc/localtime
